39 + (MediaInformation*)from: (NSString*)rawCommandOutput {
40  MediaInformation* mediaInformation = [[MediaInformation alloc] init];
41 
42  if (rawCommandOutput != nil) {
43  NSArray* split = [rawCommandOutput componentsSeparatedByString:@"\n"];
44  Boolean metadata = false;
45  Boolean sidedata = false;
46  StreamInformation *lastCreatedStream = nil;
47  NSMutableString *rawInformation = [[NSMutableString alloc] init];
48 
49  for (int i=0; i < [split count]; i++) {
50  NSString *outputLine = [split objectAtIndex:i];
51 
52  if ([outputLine hasPrefix:@"["]) {
53  metadata = false;
54  sidedata = false;
55  continue;
56  }
57 
58  NSString *trimmedLine = [outputLine stringByTrimmingCharactersInSet:[NSCharacterSet whitespaceCharacterSet]];
59 
60  if ([trimmedLine hasPrefix:@"Input"]) {
61  metadata = false;
62  sidedata = false;
63  lastCreatedStream = nil;
64 
65  void(^blockPair)(NSString **, NSString **) = [MediaInformationParser parseInputBlock:trimmedLine];
66 
67  NSString *format;
68  NSString *path;
69  blockPair(&format, &path);
70 
71  [mediaInformation setFormat:format];
72  [mediaInformation setPath:path];
73  } else if ([trimmedLine hasPrefix:@"Duration"]) {
74  metadata = false;
75  sidedata = false;
76  lastCreatedStream = nil;
77 
78  void(^blockTrio)(NSNumber **, NSNumber **, NSNumber **) = [MediaInformationParser parseDurationBlock:trimmedLine];
79 
80  NSNumber *duration;
81  NSNumber *startTime;
82  NSNumber *bitrate;
83 
84  blockTrio(&duration, &startTime, &bitrate);
85 
86  [mediaInformation setDuration:duration];
87  [mediaInformation setStartTime:startTime];
88  [mediaInformation setBitrate:bitrate];
89  } else if ([trimmedLine hasPrefix:@"Metadata"]) {
90  sidedata = false;
91  metadata = true;
92  } else if ([trimmedLine hasPrefix:@"Side data"]) {
93  metadata = false;
94  sidedata = true;
95  } else if ([trimmedLine hasPrefix:@"Stream mapping"] || [trimmedLine hasPrefix:@"Press [q] to stop"] || [trimmedLine hasPrefix:@"Output"]) {
96  break;
97  } else if ([trimmedLine hasPrefix:@"Stream"]) {
98  metadata = false;
99  sidedata = false;
100  lastCreatedStream = [MediaInformationParser parseStreamBlock:trimmedLine];
101  [mediaInformation addStream:lastCreatedStream];
102  } else if (metadata) {
103  void(^blockPair)(NSString **, NSString **) = [MediaInformationParser parseMetadataBlock:trimmedLine];
104 
105  NSString *key;
106  NSString *value;
107  blockPair(&key, &value);
108 
109  if (key != nil && value != nil) {
110  if (lastCreatedStream != nil) {
111  [lastCreatedStream addMetadata:key:value];
112  } else {
113  [mediaInformation addMetadata:key:value];
114  }
115  }
116  } else if (sidedata) {
117  void(^blockPair)(NSString **, NSString **) = [MediaInformationParser parseMetadataBlock:trimmedLine];
118 
119  NSString *key;
120  NSString *value;
121  blockPair(&key, &value);
122 
123  if (key != nil && value != nil) {
124  if (lastCreatedStream != nil) {
125  [lastCreatedStream addSidedata:key:value];
126  }
127  }
128  }
129 
130  [rawInformation appendString:outputLine];
131  [rawInformation appendString:@"\n"];
132  }
133 
134  [mediaInformation setRawInformation:rawInformation];
135  }
136 
137  return mediaInformation;
138 }

182 + (StreamInformation*)parseStreamBlock:(NSString*)input {
183  StreamInformation* streamInformation = [[StreamInformation alloc] init];
184 
185  if (input != nil) {
186  [streamInformation setIndex:[MediaInformationParser parseStreamIndex:input]];
187 
188  int typeBlockStartIndex = [MediaInformationParser index:input of:@":" from:0 times:2];
189  if ((typeBlockStartIndex > -1) && (typeBlockStartIndex < [input length])) {
190 
191  NSString *streamBlock = [input substringFromIndex:(typeBlockStartIndex + 1)];
192 
193  NSArray* parts = [streamBlock componentsSeparatedByString:@","];
194  NSString *typePart = [MediaInformationParser safeGet:parts from:0];
195  NSString *type = [MediaInformationParser parseStreamType:typePart];
196 
197  [streamInformation setType:type];
198  [streamInformation setCodec:[MediaInformationParser parseStreamCodec:typePart]];
199  [streamInformation setFullCodec:[MediaInformationParser parseStreamFullCodec:typePart]];
200 
201  NSString *part2 = [MediaInformationParser safeGet:parts from:1];
202  NSString *part3 = [MediaInformationParser safeGet:parts from:2];
203  NSString *part4 = [MediaInformationParser safeGet:parts from:3];
204  NSString *part5 = [MediaInformationParser safeGet:parts from:4];
205 
206  if ([@"video" isEqualToString:type]) {
207  int lastUsedPart = 1;
208 
209  if (part2 != nil) {
210  int pStart = [MediaInformationParser count:part2 of:@"("];
211  int pEnd = [MediaInformationParser count:part2 of:@")"];
212 
213  while (pStart != pEnd) {
214  lastUsedPart++;
215  NSString *newPart = [MediaInformationParser safeGet:parts from:lastUsedPart];
216  if (newPart == nil) {
217  break;
218  }
219  part2 = [NSString stringWithFormat:@"%@,%@", part2, newPart];
220  pStart = [MediaInformationParser count:part2 of:@"("];
221  pEnd = [MediaInformationParser count:part2 of:@")"];
222  }
223 
224  part2 = [part2 lowercaseString];
225  part2 = [part2 stringByTrimmingCharactersInSet:[NSCharacterSet whitespaceCharacterSet]];
226 
227  [streamInformation setFullFormat:part2];
228 
229  NSString *formattedPart2 = [parenthesesRegex stringByReplacingMatchesInString:part2 options:0 range:NSMakeRange(0, [part2 length]) withTemplate:@""];
230  formattedPart2 = [formattedPart2 stringByTrimmingCharactersInSet:[NSCharacterSet whitespaceCharacterSet]];
231  [streamInformation setFormat:formattedPart2];
232  }
233 
234  lastUsedPart++;
235  NSString *videoDimensionPart = [MediaInformationParser safeGet:parts from:lastUsedPart];
236  if (videoDimensionPart != nil) {
237  NSString *videoLayout = [videoDimensionPart lowercaseString];
238  videoLayout = [videoLayout stringByTrimmingCharactersInSet:[NSCharacterSet whitespaceCharacterSet]];
239 
240  void(^dimensions)(NSNumber **, NSNumber **) = [MediaInformationParser parseVideoDimensions:videoLayout];
241  NSNumber *width;
242  NSNumber *height;
243  dimensions(&width, &height);
244 
245  [streamInformation setWidth:width];
246  [streamInformation setHeight:height];
247 
248  [streamInformation setSampleAspectRatio:[MediaInformationParser parseVideoStreamSampleAspectRatio:videoLayout]];
249  [streamInformation setDisplayAspectRatio:[MediaInformationParser parseVideoStreamDisplayAspectRatio:videoLayout]];
250  }
251 
252  for (int i = lastUsedPart + 1; i < [parts count]; i++) {
253  NSString *part = [parts objectAtIndex:i];
254 
255  part = [parenthesesRegex stringByReplacingMatchesInString:part options:0 range:NSMakeRange(0, [part length]) withTemplate:@""];
256  part = [part lowercaseString];
257 
258  if ([part rangeOfString:@"kb/s"].location != NSNotFound) {
259  part = [part stringByReplacingOccurrencesOfString:@"kb/s" withString:@""];
260  part = [part stringByTrimmingCharactersInSet:[NSCharacterSet whitespaceCharacterSet]];
261  NSNumber *bitrate = [MediaInformationParser toIntegerObject:part];
262  [streamInformation setBitrate:bitrate];
263  } else if ([part rangeOfString:@"fps"].location != NSNotFound) {
264  part = [part stringByReplacingOccurrencesOfString:@"fps" withString:@""];
265  part = [part stringByTrimmingCharactersInSet:[NSCharacterSet whitespaceCharacterSet]];
266  [streamInformation setAverageFrameRate:part];
267  } else if ([part rangeOfString:@"tbr"].location != NSNotFound) {
268  part = [part stringByReplacingOccurrencesOfString:@"tbr" withString:@""];
269  part = [part stringByTrimmingCharactersInSet:[NSCharacterSet whitespaceCharacterSet]];
270  [streamInformation setRealFrameRate:part];
271  } else if ([part rangeOfString:@"tbn"].location != NSNotFound) {
272  part = [part stringByReplacingOccurrencesOfString:@"tbn" withString:@""];
273  part = [part stringByTrimmingCharactersInSet:[NSCharacterSet whitespaceCharacterSet]];
274  [streamInformation setTimeBase:part];
275  } else if ([part rangeOfString:@"tbc"].location != NSNotFound) {
276  part = [part stringByReplacingOccurrencesOfString:@"tbc" withString:@""];
277  part = [part stringByTrimmingCharactersInSet:[NSCharacterSet whitespaceCharacterSet]];
278  [streamInformation setCodecTimeBase:part];
279  }
280  }
281  } else if ([@"audio" isEqualToString:type]) {
282  if (part2 != nil) {
283  [streamInformation setSampleRate:[MediaInformationParser parseAudioStreamSampleRate:part2]];
284  }
285  if (part3 != nil) {
286  NSString *formattedPart3 = [part3 lowercaseString];
287  formattedPart3 = [formattedPart3 stringByTrimmingCharactersInSet:[NSCharacterSet whitespaceCharacterSet]];
288  [streamInformation setChannelLayout:formattedPart3];
289  }
290  if (part4 != nil) {
291  NSString *formattedPart4 = [part4 lowercaseString];
292  formattedPart4 = [formattedPart4 stringByTrimmingCharactersInSet:[NSCharacterSet whitespaceCharacterSet]];
293  [streamInformation setSampleFormat:formattedPart4];
294  }
295  if (part5 != nil) {
296  NSString *formattedPart5 = [part5 lowercaseString];
297  formattedPart5 = [parenthesesRegex stringByReplacingMatchesInString:formattedPart5 options:0 range:NSMakeRange(0, [formattedPart5 length]) withTemplate:@""];
298  formattedPart5 = [formattedPart5 stringByReplacingOccurrencesOfString:@"kb/s" withString:@""];
299  formattedPart5 = [formattedPart5 stringByTrimmingCharactersInSet:[NSCharacterSet whitespaceCharacterSet]];
300  [streamInformation setBitrate:[MediaInformationParser toIntegerObject:formattedPart5]];
301  }
302  } else if ([@"data" isEqualToString:type]) {
303  if (part2 != nil) {
304  NSString *formattedPart2 = [part2 lowercaseString];
305  formattedPart2 = [parenthesesRegex stringByReplacingMatchesInString:formattedPart2 options:0 range:NSMakeRange(0, [formattedPart2 length]) withTemplate:@""];
306  formattedPart2 = [formattedPart2 stringByReplacingOccurrencesOfString:@"kb/s" withString:@""];
307  formattedPart2 = [formattedPart2 stringByTrimmingCharactersInSet:[NSCharacterSet whitespaceCharacterSet]];
308  [streamInformation setBitrate:[MediaInformationParser toIntegerObject:formattedPart2]];
309  }
310  }
311  }
312  }
313 
314  return streamInformation;
315 }

399  } else if (mhz) {

558  int count = 1;

560

563  NSRange currentRange = [string rangeOfString:substring options:NSLiteralSearch range:NSMakeRange(searchIndex, string.length - searchIndex)];
564  if (currentRange.location != NSNotFound) {
565  startIndex = (int)currentRange.location;

568  count++;
569  }
