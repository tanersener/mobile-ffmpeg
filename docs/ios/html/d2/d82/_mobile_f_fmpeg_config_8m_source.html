226 void mobileffmpeg_log_callback_function(void *ptr, int level, const char* format, va_list vargs) {
227 
228  // DO NOT PROCESS UNWANTED LOGS
229  if (level >= 0) {
230  level &= 0xff;
231  }
232  int activeLogLevel = [MobileFFmpegConfig getLogLevel];
233 
234  // AV_LOG_STDERR logs are always redirected
235  if ((activeLogLevel == AV_LOG_QUIET && level != AV_LOG_STDERR) || (level > activeLogLevel)) {
236  return;
237  }
238 
239  NSString *logData = [[NSString alloc] initWithFormat:[NSString stringWithCString:format encoding:NSUTF8StringEncoding] arguments:vargs];
240 
241  logCallbackDataAdd(level, logData);
242 
243  // BUILDING LAST COMMAND OUTPUT AS EARLY AS POSSIBLE
244  [lastCommandOutput appendString:logData];
245 }

265 void callbackBlockFunction() {
266  int activeLogLevel = av_log_get_level();
267  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_DEBUG <= activeLogLevel)) {
268  NSLog(@"Async callback block started.\n");
269  }
270 
271  while(redirectionEnabled) {
272  @autoreleasepool {
273  @try {
274 
275  CallbackData *callbackData = callbackDataRemove();
276  if (callbackData != nil) {
277 
278  if ([callbackData getType] == LogType) {
279 
280  // LOG CALLBACK
281  int activeLogLevel = [MobileFFmpegConfig getLogLevel];
282  int levelValue = [callbackData getLogLevel];
283 
284  if ((activeLogLevel == AV_LOG_QUIET && levelValue != AV_LOG_STDERR) || (levelValue > activeLogLevel)) {
285 
286  // LOG NEITHER PRINTED NOR FORWARDED
287  } else {
288  if (logDelegate != nil) {
289 
290  // FORWARD LOG TO DELEGATE
291  [logDelegate logCallback:[callbackData getLogLevel]:[callbackData getLogData]];
292 
293  } else {
294  switch (levelValue) {
295  case AV_LOG_QUIET:
296  // PRINT NO OUTPUT
297  break;
298  default:
299  // WRITE TO NSLOG
300  NSLog(@"%@: %@", [MobileFFmpegConfig logLevelToString:[callbackData getLogLevel]], [callbackData getLogData]);
301  break;
302  }
303  }
304  }
305 
306  } else {
307 
308  // STATISTICS CALLBACK
309  Statistics *newStatistics = [[Statistics alloc] initWithVideoFrameNumber:[callbackData getStatisticsFrameNumber] fps:[callbackData getStatisticsFps] quality:[callbackData getStatisticsQuality] size:[callbackData getStatisticsSize] time:[callbackData getStatisticsTime] bitrate:[callbackData getStatisticsBitrate] speed:[callbackData getStatisticsSpeed]];
310  [lastReceivedStatistics update:newStatistics];
311 
312  if (logDelegate != nil) {
313 
314  // FORWARD STATISTICS TO DELEGATE
315  [statisticsDelegate statisticsCallback:lastReceivedStatistics];
316  }
317  }
318 
319  } else {
320  callbackWait(100);
321  }
322 
323  } @catch(NSException *exception) {
324  activeLogLevel = av_log_get_level();
325  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_WARNING <= activeLogLevel)) {
326  NSLog(@"Async callback block received error: %@n\n", exception);
327  NSLog(@"%@", [exception callStackSymbols]);
328  }
329  }
330  }
331  }
332 
333  activeLogLevel = av_log_get_level();
334  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_DEBUG <= activeLogLevel)) {
335  NSLog(@"Async callback block stopped.\n");
336  }
337 }

536 + (void)setFontDirectory: (NSString*)fontDirectoryPath with:(NSDictionary*)fontNameMapping {
537  NSError *error = nil;
538  BOOL isDirectory = YES;
539  BOOL isFile = NO;
540  int validFontNameMappingCount = 0;
541  NSString *tempConfigurationDirectory = [NSTemporaryDirectory() stringByAppendingPathComponent:@".mobileffmpeg"];
542  NSString *fontConfigurationFile = [tempConfigurationDirectory stringByAppendingPathComponent:@"fonts.conf"];
543  int activeLogLevel = av_log_get_level();
544 
545  if (![[NSFileManager defaultManager] fileExistsAtPath:tempConfigurationDirectory isDirectory:&isDirectory]) {
546 
547  if (![[NSFileManager defaultManager] createDirectoryAtPath:tempConfigurationDirectory withIntermediateDirectories:YES attributes:nil error:&error]) {
548  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_WARNING <= activeLogLevel)) {
549  NSLog(@"Failed to set font directory. Error received while creating temp conf directory: %@.", error);
550  }
551  return;
552  }
553 
554  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_DEBUG <= activeLogLevel)) {
555  NSLog(@"Created temporary font conf directory: TRUE.");
556  }
557  }
558 
559  if ([[NSFileManager defaultManager] fileExistsAtPath:fontConfigurationFile isDirectory:&isFile]) {
560  BOOL fontConfigurationDeleted = [[NSFileManager defaultManager] removeItemAtPath:fontConfigurationFile error:NULL];
561  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_DEBUG <= activeLogLevel)) {
562  NSLog(@"Deleted old temporary font configuration: %s.", fontConfigurationDeleted?"TRUE":"FALSE");
563  }
564  }
565 
566  /* PROCESS MAPPINGS FIRST */
567  NSString *fontNameMappingBlock = @"";
568  for (NSString *fontName in [fontNameMapping allKeys]) {
569  NSString *mappedFontName = [fontNameMapping objectForKey:fontName];
570 
571  if ((fontName != nil) && (mappedFontName != nil) && ([fontName length] > 0) && ([mappedFontName length] > 0)) {
572 
573  fontNameMappingBlock = [NSString stringWithFormat:@"%@\n%@\n%@%@%@\n%@\n%@\n%@%@%@\n%@\n%@\n",
574  @" <match target=\"pattern\">",
575  @" <test qual=\"any\" name=\"family\">",
576  @" <string>", fontName, @"</string>",
577  @" </test>",
578  @" <edit name=\"family\" mode=\"assign\" binding=\"same\">",
579  @" <string>", mappedFontName, @"</string>",
580  @" </edit>",
581  @" </match>"];
582 
583  validFontNameMappingCount++;
584  }
585  }
586 
587  NSString *fontConfiguration = [NSString stringWithFormat:@"%@\n%@\n%@\n%@\n%@%@%@\n%@\n%@\n",
588  @"<?xml version=\"1.0\"?>",
589  @"<!DOCTYPE fontconfig SYSTEM \"fonts.dtd\">",
590  @"<fontconfig>",
591  @" <dir>.</dir>",
592  @" <dir>", fontDirectoryPath, @"</dir>",
593  fontNameMappingBlock,
594  @"</fontconfig>"];
595 
596  if (![fontConfiguration writeToFile:fontConfigurationFile atomically:YES encoding:NSUTF8StringEncoding error:&error]) {
597  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_WARNING <= activeLogLevel)) {
598  NSLog(@"Failed to set font directory. Error received while saving font configuration: %@.", error);
599  }
600  return;
601  }
602  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_DEBUG <= activeLogLevel)) {
603  NSLog(@"Saved new temporary font configuration with %d font name mappings.", validFontNameMappingCount);
604  }
605 
606  [MobileFFmpegConfig setFontconfigConfigurationPath:tempConfigurationDirectory];
607 
608  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_DEBUG <= activeLogLevel)) {
609  NSLog(@"Font directory %@ registered successfully.", fontDirectoryPath);
610  }
611 }

809 + (NSArray*)getExternalLibraries {
810  NSString *buildConfiguration = [MobileFFmpegConfig getBuildConf];
811  NSMutableArray *enabledLibraryArray = [[NSMutableArray alloc] init];
812 
813  for (int i=0; i < [supportedExternalLibraries count]; i++) {
814  NSString *supportedExternalLibrary = [supportedExternalLibraries objectAtIndex:i];
815 
816  NSString *libraryName1 = [NSString stringWithFormat:@"enable-%@", supportedExternalLibrary];
817  NSString *libraryName2 = [NSString stringWithFormat:@"enable-lib%@", supportedExternalLibrary];
818 
819  if ([buildConfiguration rangeOfString:libraryName1].location != NSNotFound || [buildConfiguration rangeOfString:libraryName2].location != NSNotFound) {
820  [enabledLibraryArray addObject:supportedExternalLibrary];
821  }
822  }
823 
824  [enabledLibraryArray sortUsingSelector:@selector(compare:)];
