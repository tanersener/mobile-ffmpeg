249 void callbackBlockFunction() {
250  NSLog(@"Async callback block started.\n");
251 
252  while(redirectionEnabled) {
253  @autoreleasepool {
254  @try {
255 
256  CallbackData *callbackData = callbackDataRemove();
257  if (callbackData != nil) {
258 
259  if ([callbackData getType] == LogType) {
260 
261  // LOG CALLBACK
262  int activeLogLevel = [MobileFFmpegConfig getLogLevel];
263 
264  if (runningSystemCommand == 1) {
265 
266  // REDIRECT SYSTEM OUTPUT
267  if ((activeLogLevel != AV_LOG_QUIET) && ([callbackData getLogLevel] <= activeLogLevel)) {
268  [systemCommandOutput appendString:[callbackData getLogData]];
269  }
270  } else if ((activeLogLevel == AV_LOG_QUIET) || ([callbackData getLogLevel] > activeLogLevel)) {
271 
272  // LOG NEITHER PRINTED NOR FORWARDED
273  } else {
274 
275  // ALWAYS REDIRECT COMMAND OUTPUT
276  [lastCommandOutput appendString:[callbackData getLogData]];
277 
278  if (logDelegate != nil) {
279 
280  // FORWARD LOG TO DELEGATE
281  [logDelegate logCallback:[callbackData getLogLevel]:[callbackData getLogData]];
282 
283  } else {
284 
285  // WRITE TO NSLOG
286  NSLog(@"%@: %@", [MobileFFmpegConfig logLevelToString:[callbackData getLogLevel]], [callbackData getLogData]);
287  }
288 
289  }
290 
291  } else {
292 
293  // STATISTICS CALLBACK
294  Statistics *newStatistics = [[Statistics alloc] initWithVideoFrameNumber:[callbackData getStatisticsFrameNumber] fps:[callbackData getStatisticsFps] quality:[callbackData getStatisticsQuality] size:[callbackData getStatisticsSize] time:[callbackData getStatisticsTime] bitrate:[callbackData getStatisticsBitrate] speed:[callbackData getStatisticsSpeed]];
295  [lastReceivedStatistics update:newStatistics];
296 
297  if (logDelegate != nil) {
298 
299  // FORWARD STATISTICS TO DELEGATE
300  [statisticsDelegate statisticsCallback:lastReceivedStatistics];
301  }
302  }
303 
304  } else {
305  callbackWait(100);
306  }
307 
308  } @catch(NSException *exception) {
309  NSLog(@"Async callback block received error: %@n\n", exception);
310  NSLog(@"%@", [exception callStackSymbols]);
311  }
312  }
313  }
314 
315  NSLog(@"Async callback block stopped.\n");
316 }

337 int mobileffmpeg_system_execute(NSArray *arguments, NSArray *commandOutputEndPatternList, long timeout) {
338  systemCommandOutput = [[NSMutableString alloc] init];
339  runningSystemCommand = 1;
340 
341  int rc = [MobileFFmpeg executeWithArguments:arguments];
342 
343  long totalWaitTime = 0;
344 
345  while ((systemCommandOutputContainsPattern(commandOutputEndPatternList) == 0) && (totalWaitTime < timeout)) {
346  [NSThread sleepForTimeInterval:.02];
347  totalWaitTime += 20;
348  }
349 
350  runningSystemCommand = 0;
351 
352  [MobileFFmpeg cancel];
353 
354  return rc;

554 + (void)setFontDirectory: (NSString*)fontDirectoryPath with:(NSDictionary*)fontNameMapping {
555  NSError *error = nil;
556  BOOL isDirectory = YES;
557  BOOL isFile = NO;
558  int validFontNameMappingCount = 0;
559  NSString *tempConfigurationDirectory = [NSTemporaryDirectory() stringByAppendingPathComponent:@".mobileffmpeg"];
560  NSString *fontConfigurationFile = [tempConfigurationDirectory stringByAppendingPathComponent:@"fonts.conf"];
561 
562  if (![[NSFileManager defaultManager] fileExistsAtPath:tempConfigurationDirectory isDirectory:&isDirectory]) {
563 
564  if (![[NSFileManager defaultManager] createDirectoryAtPath:tempConfigurationDirectory withIntermediateDirectories:YES attributes:nil error:&error]) {
565  NSLog(@"Failed to set font directory. Error received while creating temp conf directory: %@.", error);
566  return;
567  }
568 
569  NSLog(@"Created temporary font conf directory: TRUE.");
570  }
571 
572  if ([[NSFileManager defaultManager] fileExistsAtPath:fontConfigurationFile isDirectory:&isFile]) {
573  BOOL fontConfigurationDeleted = [[NSFileManager defaultManager] removeItemAtPath:fontConfigurationFile error:NULL];
574  NSLog(@"Deleted old temporary font configuration: %s.", fontConfigurationDeleted?"TRUE":"FALSE");
575  }
576 
577  /* PROCESS MAPPINGS FIRST */
578  NSString *fontNameMappingBlock = @"";
579  for (NSString *fontName in [fontNameMapping allKeys]) {
580  NSString *mappedFontName = [fontNameMapping objectForKey:fontName];
581 
582  if ((fontName != nil) && (mappedFontName != nil) && ([fontName length] > 0) && ([mappedFontName length] > 0)) {
583 
584  fontNameMappingBlock = [NSString stringWithFormat:@"%@\n%@\n%@%@%@\n%@\n%@\n%@%@%@\n%@\n%@\n",
585  @" <match target=\"pattern\">",
586  @" <test qual=\"any\" name=\"family\">",
587  @" <string>", fontName, @"</string>",
588  @" </test>",
589  @" <edit name=\"family\" mode=\"assign\" binding=\"same\">",
590  @" <string>", mappedFontName, @"</string>",
591  @" </edit>",
592  @" </match>"];
593 
594  validFontNameMappingCount++;
595  }
596  }
597 
598  NSString *fontConfiguration = [NSString stringWithFormat:@"%@\n%@\n%@\n%@\n%@%@%@\n%@\n%@\n",
599  @"<?xml version=\"1.0\"?>",
600  @"<!DOCTYPE fontconfig SYSTEM \"fonts.dtd\">",
601  @"<fontconfig>",
602  @" <dir>.</dir>",
603  @" <dir>", fontDirectoryPath, @"</dir>",
604  fontNameMappingBlock,
605  @"</fontconfig>"];
606 
607  if (![fontConfiguration writeToFile:fontConfigurationFile atomically:YES encoding:NSUTF8StringEncoding error:&error]) {
608  NSLog(@"Failed to set font directory. Error received while saving font configuration: %@.", error);
609  return;
610  }
611  NSLog(@"Saved new temporary font configuration with %d font name mappings.", validFontNameMappingCount);
612 
613  [MobileFFmpegConfig setFontconfigConfigurationPath:tempConfigurationDirectory];
614 
615  NSLog(@"Font directory %@ registered successfully.", fontDirectoryPath);
