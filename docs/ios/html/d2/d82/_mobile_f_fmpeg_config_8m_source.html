222 void mobileffmpeg_log_callback_function(void *ptr, int level, const char* format, va_list vargs) {
223 
224  // DO NOT PROCESS UNWANTED LOGS
225  if (level >= 0) {
226  level &= 0xff;
227  }
228  int activeLogLevel = [MobileFFmpegConfig getLogLevel];
229  if ((activeLogLevel == AV_LOG_QUIET) || (level > activeLogLevel)) {
230  return;
231  }
232 
233  NSString *logData = [[NSString alloc] initWithFormat:[NSString stringWithCString:format encoding:NSUTF8StringEncoding] arguments:vargs];
234 
235  logCallbackDataAdd(level, logData);
236 
237  // BUILDING LAST COMMAND OUTPUT AS EARLY AS POSSIBLE
238  [lastCommandOutput appendString:logData];
239 }

259 void callbackBlockFunction() {
260  int activeLogLevel = av_log_get_level();
261  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_DEBUG <= activeLogLevel)) {
262  NSLog(@"Async callback block started.\n");
263  }
264 
265  while(redirectionEnabled) {
266  @autoreleasepool {
267  @try {
268 
269  CallbackData *callbackData = callbackDataRemove();
270  if (callbackData != nil) {
271 
272  if ([callbackData getType] == LogType) {
273 
274  // LOG CALLBACK
275  int activeLogLevel = [MobileFFmpegConfig getLogLevel];
276 
277  if (runningSystemCommand == 1) {
278 
279  // REDIRECT SYSTEM OUTPUT
280  if ((activeLogLevel != AV_LOG_QUIET) && ([callbackData getLogLevel] <= activeLogLevel)) {
281  [systemCommandOutput appendString:[callbackData getLogData]];
282  }
283  } else if ((activeLogLevel == AV_LOG_QUIET) || ([callbackData getLogLevel] > activeLogLevel)) {
284 
285  // LOG NEITHER PRINTED NOR FORWARDED
286  } else {
287  if (logDelegate != nil) {
288 
289  // FORWARD LOG TO DELEGATE
290  [logDelegate logCallback:[callbackData getLogLevel]:[callbackData getLogData]];
291 
292  } else {
293 
294  // WRITE TO NSLOG
295  NSLog(@"%@: %@", [MobileFFmpegConfig logLevelToString:[callbackData getLogLevel]], [callbackData getLogData]);
296  }
297 
298  }
299 
300  } else {
301 
302  // STATISTICS CALLBACK
303  Statistics *newStatistics = [[Statistics alloc] initWithVideoFrameNumber:[callbackData getStatisticsFrameNumber] fps:[callbackData getStatisticsFps] quality:[callbackData getStatisticsQuality] size:[callbackData getStatisticsSize] time:[callbackData getStatisticsTime] bitrate:[callbackData getStatisticsBitrate] speed:[callbackData getStatisticsSpeed]];
304  [lastReceivedStatistics update:newStatistics];
305 
306  if (logDelegate != nil) {
307 
308  // FORWARD STATISTICS TO DELEGATE
309  [statisticsDelegate statisticsCallback:lastReceivedStatistics];
310  }
311  }
312 
313  } else {
314  callbackWait(100);
315  }
316 
317  } @catch(NSException *exception) {
318  activeLogLevel = av_log_get_level();
319  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_WARNING <= activeLogLevel)) {
320  NSLog(@"Async callback block received error: %@n\n", exception);
321  NSLog(@"%@", [exception callStackSymbols]);
322  }
323  }
324  }
325  }
326 
327  activeLogLevel = av_log_get_level();
328  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_DEBUG <= activeLogLevel)) {
329  NSLog(@"Async callback block stopped.\n");
330  }
331 }

353 int mobileffmpeg_system_execute(NSArray *arguments, NSMutableArray *commandOutputEndPatternList, NSString *successPattern, long timeout) {
354  if (successPattern != nil) {
355  [commandOutputEndPatternList addObject:successPattern];
356  }
357  systemCommandOutput = [[NSMutableString alloc] init];
358  runningSystemCommand = 1;
359 
360  int rc = [MobileFFmpeg executeWithArguments:arguments];
361 
362  long totalWaitTime = 0;
363 
364  while ((systemCommandOutputContainsPattern(commandOutputEndPatternList) == 0) && (totalWaitTime < timeout)) {
365  [NSThread sleepForTimeInterval:.02];
366  totalWaitTime += 20;
367  }
368 
369  runningSystemCommand = 0;
370 
371  [MobileFFmpeg cancel];
372 
373  if (successPattern != nil && [systemCommandOutput containsString:successPattern]) {
374  return 0;
375  } else {
376  return rc;
377  }

576 + (void)setFontDirectory: (NSString*)fontDirectoryPath with:(NSDictionary*)fontNameMapping {
577  NSError *error = nil;
578  BOOL isDirectory = YES;
579  BOOL isFile = NO;
580  int validFontNameMappingCount = 0;
581  NSString *tempConfigurationDirectory = [NSTemporaryDirectory() stringByAppendingPathComponent:@".mobileffmpeg"];
582  NSString *fontConfigurationFile = [tempConfigurationDirectory stringByAppendingPathComponent:@"fonts.conf"];
583  int activeLogLevel = av_log_get_level();
584 
585  if (![[NSFileManager defaultManager] fileExistsAtPath:tempConfigurationDirectory isDirectory:&isDirectory]) {
586 
587  if (![[NSFileManager defaultManager] createDirectoryAtPath:tempConfigurationDirectory withIntermediateDirectories:YES attributes:nil error:&error]) {
588  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_WARNING <= activeLogLevel)) {
589  NSLog(@"Failed to set font directory. Error received while creating temp conf directory: %@.", error);
590  }
591  return;
592  }
593 
594  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_DEBUG <= activeLogLevel)) {
595  NSLog(@"Created temporary font conf directory: TRUE.");
596  }
597  }
598 
599  if ([[NSFileManager defaultManager] fileExistsAtPath:fontConfigurationFile isDirectory:&isFile]) {
600  BOOL fontConfigurationDeleted = [[NSFileManager defaultManager] removeItemAtPath:fontConfigurationFile error:NULL];
601  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_DEBUG <= activeLogLevel)) {
602  NSLog(@"Deleted old temporary font configuration: %s.", fontConfigurationDeleted?"TRUE":"FALSE");
603  }
604  }
605 
606  /* PROCESS MAPPINGS FIRST */
607  NSString *fontNameMappingBlock = @"";
608  for (NSString *fontName in [fontNameMapping allKeys]) {
609  NSString *mappedFontName = [fontNameMapping objectForKey:fontName];
610 
611  if ((fontName != nil) && (mappedFontName != nil) && ([fontName length] > 0) && ([mappedFontName length] > 0)) {
612 
613  fontNameMappingBlock = [NSString stringWithFormat:@"%@\n%@\n%@%@%@\n%@\n%@\n%@%@%@\n%@\n%@\n",
614  @" <match target=\"pattern\">",
615  @" <test qual=\"any\" name=\"family\">",
616  @" <string>", fontName, @"</string>",
617  @" </test>",
618  @" <edit name=\"family\" mode=\"assign\" binding=\"same\">",
619  @" <string>", mappedFontName, @"</string>",
620  @" </edit>",
621  @" </match>"];
622 
623  validFontNameMappingCount++;
624  }
625  }
626 
627  NSString *fontConfiguration = [NSString stringWithFormat:@"%@\n%@\n%@\n%@\n%@%@%@\n%@\n%@\n",
628  @"<?xml version=\"1.0\"?>",
629  @"<!DOCTYPE fontconfig SYSTEM \"fonts.dtd\">",
630  @"<fontconfig>",
631  @" <dir>.</dir>",
632  @" <dir>", fontDirectoryPath, @"</dir>",
633  fontNameMappingBlock,
634  @"</fontconfig>"];
635 
636  if (![fontConfiguration writeToFile:fontConfigurationFile atomically:YES encoding:NSUTF8StringEncoding error:&error]) {
637  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_WARNING <= activeLogLevel)) {
638  NSLog(@"Failed to set font directory. Error received while saving font configuration: %@.", error);
639  }
640  return;
641  }
642  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_DEBUG <= activeLogLevel)) {
643  NSLog(@"Saved new temporary font configuration with %d font name mappings.", validFontNameMappingCount);
644  }
645 
646  [MobileFFmpegConfig setFontconfigConfigurationPath:tempConfigurationDirectory];
647 
648  if ((activeLogLevel != AV_LOG_QUIET) && (AV_LOG_DEBUG <= activeLogLevel)) {
649  NSLog(@"Font directory %@ registered successfully.", fontDirectoryPath);
650  }
651 }
