127 static void avutil_log_format_line(void *avcl, int level, const char *fmt, va_list vl, AVBPrint part[4], int *print_prefix) {
128  int flags = av_log_get_flags();
129  AVClass* avc = avcl ? *(AVClass **) avcl : NULL;
130  av_bprint_init(part+0, 0, 1);
131  av_bprint_init(part+1, 0, 1);
132  av_bprint_init(part+2, 0, 1);
133  av_bprint_init(part+3, 0, 65536);
134 
135  if (*print_prefix && avc) {
136  if (avc->parent_log_context_offset) {
137  AVClass** parent = *(AVClass ***) (((uint8_t *) avcl) +
138  avc->parent_log_context_offset);
139  if (parent && *parent) {
140  av_bprintf(part+0, "[%s @ %p] ",
141  (*parent)->item_name(parent), parent);
142  }
143  }
144  av_bprintf(part+1, "[%s @ %p] ",
145  avc->item_name(avcl), avcl);
146  }
147 
148  if (*print_prefix && (level > AV_LOG_QUIET) && (flags & AV_LOG_PRINT_LEVEL))
149  av_bprintf(part+2, "[%s] ", avutil_log_get_level_str(level));
150 
151  av_vbprintf(part+3, fmt, vl);
152 
153  if(*part[0].str || *part[1].str || *part[2].str || *part[3].str) {
154  char lastc = part[3].len && part[3].len <= part[3].size ? part[3].str[part[3].len - 1] : 0;
155  *print_prefix = lastc == '\n' || lastc == '\r';
156  }
157 }

209 void monitorWait(int milliSeconds) {
210  struct timeval tp;
211  struct timespec ts;
212  int rc;
213 
214  rc = gettimeofday(&tp, NULL);
215  if (rc) {
216  return;
217  }
218 
219  ts.tv_sec = tp.tv_sec;
220  ts.tv_nsec = tp.tv_usec * 1000;
221  ts.tv_sec += milliSeconds / 1000;
222  ts.tv_nsec += (milliSeconds % 1000)*1000000;
223 
224  pthread_mutex_lock(&monitorMutex);
225  pthread_cond_timedwait(&monitorCondition, &monitorMutex, &ts);
226  pthread_mutex_unlock(&monitorMutex);
227 }

238 void logCallbackDataAdd(int level, const char *data) {
239 
240  // CREATE DATA STRUCT FIRST
241  struct CallbackData *newData = (struct CallbackData*)malloc(sizeof(struct CallbackData));
242  newData->type = 1;
243  newData->logLevel = level;
244  size_t dataSize = strlen(data) + 1;
245  newData->logData = (char*)malloc(dataSize);
246  memcpy(newData->logData, data, dataSize);
247  newData->next = NULL;
248 
249  mutexLock();
250 
251  // INSERT IT TO THE END OF QUEUE
252  if (callbackDataTail == NULL) {
253  callbackDataTail = newData;
254 
255  if (callbackDataHead != NULL) {
256  LOGE("Dangling callback data head detected. This can cause memory leak.");
257  } else {
258  callbackDataHead = newData;
259  }
260  } else {
261  struct CallbackData *oldTail = callbackDataTail;
262  oldTail->next = newData;
263 
264  callbackDataTail = newData;
265  }
266 
267  mutexUnlock();
268 
269  monitorNotify();
270 }

316 struct CallbackData *callbackDataRemove() {
317  struct CallbackData *currentData;
318 
319  mutexLock();
320 
321  if (callbackDataHead == NULL) {
322  currentData = NULL;
323  } else {
324  currentData = callbackDataHead;
325 
326  struct CallbackData *nextHead = currentData->next;
327  if (nextHead == NULL) {
328  if (callbackDataHead != callbackDataTail) {
329  LOGE("Head and tail callback data pointers do not match for single callback data element. This can cause memory leak.");
330  } else {
331  callbackDataTail = NULL;
332  }
333  callbackDataHead = NULL;
334 
335  } else {
336  callbackDataHead = nextHead;
337  }
338  }
339 
340  mutexUnlock();
341 
342  return currentData;
343 }

353 void mobileffmpeg_log_callback_function(void *ptr, int level, const char* format, va_list vargs) {
354  char line[LOG_LINE_SIZE];
355  AVBPrint part[4];
356  int print_prefix = 1;
357 
358  if (level >= 0) {
359  level &= 0xff;
360  }
361  int activeLogLevel = av_log_get_level();
362  if ((activeLogLevel == AV_LOG_QUIET) || (level > activeLogLevel)) {
363  return;
364  }
365 
366  avutil_log_format_line(ptr, level, format, vargs, part, &print_prefix);
367  snprintf(line, sizeof(line), "%s%s%s%s", part[0].str, part[1].str, part[2].str, part[3].str);
368 
369  avutil_log_sanitize(part[0].str);
370  logCallbackDataAdd(level, part[0].str);
371  avutil_log_sanitize(part[1].str);
372  logCallbackDataAdd(level, part[1].str);
373  avutil_log_sanitize(part[2].str);
374  logCallbackDataAdd(level, part[2].str);
375  avutil_log_sanitize(part[3].str);
376  logCallbackDataAdd(level, part[3].str);
377 
378  av_bprint_finalize(part+3, NULL);
379 }

399 void *callbackThreadFunction() {
400  JNIEnv *env;
401  jint getEnvRc = (*globalVm)->GetEnv(globalVm, (void**) &env, JNI_VERSION_1_6);
402  if (getEnvRc != JNI_OK) {
403  if (getEnvRc != JNI_EDETACHED) {
404  LOGE("Callback thread failed to GetEnv for class %s with rc %d.\n", configClassName, getEnvRc);
405  return NULL;
406  }
407 
408  if ((*globalVm)->AttachCurrentThread(globalVm, &env, NULL) != 0) {
409  LOGE("Callback thread failed to AttachCurrentThread for class %s.\n", configClassName);
410  return NULL;
411  }
412  }
413 
414  LOGD("Callback thread started.\n");
415 
416  while(redirectionEnabled) {
417 
418  struct CallbackData *callbackData = callbackDataRemove();
419  if (callbackData != NULL) {
420  if (callbackData->type == 1) {
421 
422  // LOG CALLBACK
423 
424  size_t size = strlen(callbackData->logData);
425 
426  jbyteArray byteArray = (jbyteArray) (*env)->NewByteArray(env, size);
427  (*env)->SetByteArrayRegion(env, byteArray, 0, size, (jbyte *)callbackData->logData);
428  (*env)->CallStaticVoidMethod(env, configClass, logMethod, callbackData->logLevel, byteArray);
429  (*env)->DeleteLocalRef(env, byteArray);
430 
431  // CLEAN LOG DATA
432  free(callbackData->logData);
433 
434  } else {
435 
436  // STATISTICS CALLBACK
437 
438  (*env)->CallStaticVoidMethod(env, configClass, statisticsMethod,
439  callbackData->statisticsFrameNumber, callbackData->statisticsFps,
440  callbackData->statisticsQuality, callbackData->statisticsSize,
441  callbackData->statisticsTime, callbackData->statisticsBitrate,
442  callbackData->statisticsSpeed);
443 
444  }
445 
446  // CLEAN STRUCT
447  callbackData->next = NULL;
448  free(callbackData);
449 
450  } else {
451  monitorWait(100);
452  }
453  }
454 
455  (*globalVm)->DetachCurrentThread(globalVm);
456 
457  LOGD("Callback thread stopped.\n");
458 
459  return NULL;
460 }

469 jint JNI_OnLoad(JavaVM *vm, void *reserved) {
470  JNIEnv *env;
471  if ((*vm)->GetEnv(vm, (void**)(&env), JNI_VERSION_1_6) != JNI_OK) {
472  LOGE("OnLoad failed to GetEnv for class %s.\n", configClassName);
473  return JNI_FALSE;
474  }
475 
476  jclass localConfigClass = (*env)->FindClass(env, configClassName);
477  if (localConfigClass == NULL) {
478  LOGE("OnLoad failed to FindClass %s.\n", configClassName);
479  return JNI_FALSE;
480  }
481 
482  if ((*env)->RegisterNatives(env, localConfigClass, configMethods, 9) < 0) {
483  LOGE("OnLoad failed to RegisterNatives for class %s.\n", configClassName);
484  return JNI_FALSE;
485  }
486 
487  (*env)->GetJavaVM(env, &globalVm);
488 
489  logMethod = (*env)->GetStaticMethodID(env, localConfigClass, "log", "(I[B)V");
490  if (logMethod == NULL) {
491  LOGE("OnLoad thread failed to GetMethodID for %s.\n", "log");
492  (*globalVm)->DetachCurrentThread(globalVm);
493  return JNI_FALSE;
494  }
495 
496  statisticsMethod = (*env)->GetStaticMethodID(env, localConfigClass, "statistics", "(IFFJIDD)V");
497  if (logMethod == NULL) {
498  LOGE("OnLoad thread failed to GetMethodID for %s.\n", "statistics");
499  (*globalVm)->DetachCurrentThread(globalVm);
500  return JNI_FALSE;
501  }
502 
503  configClass = (jclass) ((*env)->NewGlobalRef(env, localConfigClass));
504 
505  redirectionEnabled = 0;
506 
507  callbackDataHead = NULL;
508  callbackDataTail = NULL;
509 
510  mutexInit();
511  monitorInit();
512 
513  return JNI_VERSION_1_6;
514 }
